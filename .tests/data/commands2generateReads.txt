(wgsim command is in the conda environment 'test')
1. for single M. genitalium G37 genome (ASM2732v1) at 5X coverage
wgsim -N 23019 -r 0 -R 0 -X 0 -1 126 -2 126 GCF_000027325.1_ASM2732v1_genomic.fna ASM2732v1_1.fastq ASM2732v1_2.fastq && \
sed -E "s/\/(1|2)//" ASM2732v1_1.fastq > ASM2732v1_fixed_R1.fastq && \
sed -E "s/\/(1|2)//" ASM2732v1_2.fastq > ASM2732v1_fixed_R2.fastq && \
gzip ASM2732v1_fixed_R1.fastq && \
gzip ASM2732v1_fixed_R2.fastq

2. for metagenomic sample with M. genitalium G37 genome (ASM2732v1) and B. aphidicola Meu genome (ASM523729v1) at 5X coverage
echo ">Mixed_genome" > Mixed_genome.fna && \
tail -n +2 GCF_000027325.1_ASM2732v1_genomic.fna >> Mixed_genome.fna && \
tail -n +2 GCF_005237295.1_ASM523729v1_genomic.fna >> Mixed_genome.fna && \
wgsim -N 49235 -r 0 -R 0 -X 0 -1 126 -2 126 Mixed_genome.fna Mixed_1.fastq Mixed_2.fastq && \
sed -E "s/\/(1|2)//" Mixed_1.fastq > Mixed_fixed_R1.fastq && \
sed -E "s/\/(1|2)//" Mixed_2.fastq > Mixed_fixed_R2.fastq && \
gzip Mixed_fixed_R1.fastq && \
gzip Mixed_fixed_R2.fastq
